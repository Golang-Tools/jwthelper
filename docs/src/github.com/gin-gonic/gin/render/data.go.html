// Data contains ContentType and bytes data.
type Data struct {
	ContentType string
	Data        []byte
}

// Render (Data) writes data with custom ContentType.
func ( Data) ( http.ResponseWriter) ( error) {
	.WriteContentType()
	_,  = .Write(.Data)
	return
}
