const bincDoPrune = true // No longer needed. Needed before as C lib did not support pruning.

// vd as low 4 bits (there are 16 slots)
const (
	bincVdSpecial byte = iota
	bincVdPosInt
	bincVdNegInt
	bincVdFloat

	bincVdString
	bincVdByteArray
	bincVdArray
	bincVdMap

	bincVdTimestamp
	bincVdSmallInt
	bincVdUnicodeOther
	bincVdSymbol

	bincVdDecimal
	_               // open slot
	_               // open slot
	bincVdCustomExt = 0x0f
)

const (
	bincSpNil byte = iota
	bincSpFalse
	bincSpTrue
	bincSpNan
	bincSpPosInf
	bincSpNegInf
	bincSpZeroFloat
	bincSpZero
	bincSpNegOne
)

const (
	bincFlBin16 byte = iota
	bincFlBin32
	_ // bincFlBin32e
	bincFlBin64
	_ // bincFlBin64e
	// others not currently supported
)

func (,  byte) string {
	switch  {
	case bincVdSpecial:
		switch  {
		case bincSpNil:
			return "nil"
		case bincSpFalse:
			return "false"
		case bincSpTrue:
			return "true"
		case bincSpNan, bincSpPosInf, bincSpNegInf, bincSpZeroFloat:
			return "float"
		case bincSpZero:
			return "uint"
		case bincSpNegOne:
			return "int"
		default:
			return "unknown"
		}
	case bincVdSmallInt, bincVdPosInt:
		return "uint"
	case bincVdNegInt:
		return "int"
	case bincVdFloat:
		return "float"
	case bincVdSymbol:
		return "string"
	case bincVdString:
		return "string"
	case bincVdByteArray:
		return "bytes"
	case bincVdTimestamp:
		return "time"
	case bincVdCustomExt:
		return "ext"
	case bincVdArray:
		return "array"
	case bincVdMap:
		return "map"
	default:
		return "unknown"
	}
}

type bincEncDriver struct {
	noBuiltInTypes
	encDriverNoopContainerWriter
	h *BincHandle
	m map[string]uint16 // symbols
	b [8]byte           // scratch, used for encoding numbers - bigendian style
	s uint16            // symbols sequencer
	_ [4]uint64         // padding
	e Encoder
}

func ( *bincEncDriver) () *Encoder {
	return &.e
}

func ( *bincEncDriver) () {
	.e.encWr.writen1(bincVdSpecial<<4 | bincSpNil)
}

func ( *bincEncDriver) ( time.Time) {
	if .IsZero() {
		.EncodeNil()
	} else {
		 := bincEncodeTime()
		.e.encWr.writen1(bincVdTimestamp<<4 | uint8(len()))
		.e.encWr.writeb()
	}
}

func ( *bincEncDriver) ( bool) {
	if  {
		.e.encWr.writen1(bincVdSpecial<<4 | bincSpTrue)
	} else {
		.e.encWr.writen1(bincVdSpecial<<4 | bincSpFalse)
	}
}

func ( *bincEncDriver) ( float32) {
	if  == 0 {
		.e.encWr.writen1(bincVdSpecial<<4 | bincSpZeroFloat)
		return
	}
	.e.encWr.writen1(bincVdFloat<<4 | bincFlBin32)
	bigenHelper{.b[:4], .e.w()}.writeUint32(math.Float32bits())
}

func ( *bincEncDriver) ( float64) {
	if  == 0 {
		.e.encWr.writen1(bincVdSpecial<<4 | bincSpZeroFloat)
		return
	}
	bigen.PutUint64(.b[:8], math.Float64bits())
	if bincDoPrune {
		 := 7
		for ;  >= 0 && (.b[] == 0); -- {
		}
		++
		if  <= 6 {
			.e.encWr.writen1(bincVdFloat<<4 | 0x8 | bincFlBin64)
			.e.encWr.writen1(byte())
			.e.encWr.writeb(.b[:])
			return
		}
	}
	.e.encWr.writen1(bincVdFloat<<4 | bincFlBin64)
	.e.encWr.writeb(.b[:8])
}

func ( *bincEncDriver) ( byte,  bool,  uint64,  uint8) {
	if  == 4 {
		bigen.PutUint32(.b[:], uint32())
	} else {
		bigen.PutUint64(.b[:], )
	}
	if bincDoPrune {
		 := pruneSignExt(.b[:], )
		.e.encWr.writen1( |  - 1 - byte())
		.e.encWr.writeb(.b[:])
	} else {
		.e.encWr.writen1( |  - 1)
		.e.encWr.writeb(.b[:])
	}
}

func ( *bincEncDriver) ( int64) {
	// const nbd byte = bincVdNegInt << 4
	if  >= 0 {
		.encUint(bincVdPosInt<<4, true, uint64())
	} else if  == -1 {
		.e.encWr.writen1(bincVdSpecial<<4 | bincSpNegOne)
	} else {
		.encUint(bincVdNegInt<<4, false, uint64(-))
	}
}

func ( *bincEncDriver) ( uint64) {

}

func ( *bincEncDriver) ( byte,  bool,  uint64) {
	if  == 0 {
		.e.encWr.writen1(bincVdSpecial<<4 | bincSpZero)
	} else if  &&  >= 1 &&  <= 16 {
		.e.encWr.writen1(bincVdSmallInt<<4 | byte(-1))
	} else if  <= math.MaxUint8 {
		.e.encWr.writen2(|0x0, byte())
	} else if  <= math.MaxUint16 {
		.e.encWr.writen1( | 0x01)
		bigenHelper{.b[:2], .e.w()}.writeUint16(uint16())
	} else if  <= math.MaxUint32 {
		.encIntegerPrune(, , , 4)
	} else {
		.encIntegerPrune(, , , 8)
	}
}

type bincDecDriver struct {
	decDriverNoopContainerReader
	noBuiltInTypes

	h      *BincHandle
	bdRead bool
	bd     byte
	vd     byte
	vs     byte

	fnil bool
	// _      [3]byte // padding
	// linear searching on this slice is ok,
	// because we typically expect < 32 symbols in each stream.
	s map[uint16][]byte // []bincDecSymbol

	b [8]byte   // scratch for decoding numbers - big endian style
	_ [4]uint64 // padding cache-aligned

	d Decoder
}

func ( *bincDecDriver) () *Decoder {
	return &.d
}

func ( *bincDecDriver) () {
	.bd = .d.decRd.readn1()
	.vd = .bd >> 4
	.vs = .bd & 0x0f
	.bdRead = true
}

func ( *bincDecDriver) () {
	if .bdRead {
		.d.decRd.unreadn1()
		.bdRead = false
	}
}

func ( *bincDecDriver) () ( bool) {
	.fnil = false
	if !.bdRead {
		.readNextBd()
	}
	if .bd == bincVdSpecial<<4|bincSpNil {
		.bdRead = false
		.fnil = true
		 = true
	}
	return
}

func ( *bincDecDriver) () bool {
	return .fnil
}

func ( *bincDecDriver) () bool {
	return .advanceNil()
}

func ( *bincDecDriver) () ( valueType) {
	if !.bdRead {
		.readNextBd()
	}
	.fnil = false
	// if d.vd == bincVdSpecial && d.vs == bincSpNil {
	if .bd == bincVdSpecial<<4|bincSpNil {
		.bdRead = false
		.fnil = true
		return valueTypeNil
	} else if .vd == bincVdByteArray {
		return valueTypeBytes
	} else if .vd == bincVdString {
		return valueTypeString
	} else if .vd == bincVdArray {
		return valueTypeArray
	} else if .vd == bincVdMap {
		return valueTypeMap
	}
	return valueTypeUnset
}

func ( *bincDecDriver) () ( time.Time) {
	if .advanceNil() {
		return
	}
	if .vd != bincVdTimestamp {
		.d.errorf("cannot decode time - %s %x-%x/%s", msgBadDesc, .vd, .vs, bincdesc(.vd, .vs))
		return
	}
	,  := bincDecodeTime(.d.decRd.readx(uint(.vs)))
	if  != nil {
		panic()
	}
	.bdRead = false
	return
}

func ( *bincDecDriver) (,  byte) {
	if &0x8 == 0 {
		.d.decRd.readb(.b[0:])
	} else {
		 := .d.decRd.readn1()
		if  > 8 {
			.d.errorf("cannot read float - at most 8 bytes used to represent float - received %v bytes", )
			return
		}
		for  := ;  < 8; ++ {
			.b[] = 0
		}
		.d.decRd.readb(.b[0:])
	}
}

func ( *bincDecDriver) () ( float64) {
	//if true { f = math.Float64frombits(bigen.Uint64(d.d.decRd.readx(8))); break; }
	if  := .vs & 0x7;  == bincFlBin32 {
		.decFloatPre(.vs, 4)
		 = float64(math.Float32frombits(bigen.Uint32(.b[0:4])))
	} else if  == bincFlBin64 {
		.decFloatPre(.vs, 8)
		 = math.Float64frombits(bigen.Uint64(.b[0:8]))
	} else {
		.d.errorf("read float - only float32 and float64 are supported - %s %x-%x/%s",
			msgBadDesc, .vd, .vs, bincdesc(.vd, .vs))
		return
	}
	return
}

func ( *bincDecDriver) () ( uint64) {
	// need to inline the code (interface conversion and type assertion expensive)
	switch .vs {
	case 0:
		 = uint64(.d.decRd.readn1())
	case 1:
		.d.decRd.readb(.b[6:8])
		 = uint64(bigen.Uint16(.b[6:8]))
	case 2:
		.b[4] = 0
		.d.decRd.readb(.b[5:8])
		 = uint64(bigen.Uint32(.b[4:8]))
	case 3:
		.d.decRd.readb(.b[4:8])
		 = uint64(bigen.Uint32(.b[4:8]))
	case 4, 5, 6:
		 := 7 - .vs
		.d.decRd.readb(.b[:8])
		for  := uint8(0);  < ; ++ {
			.b[] = 0
		}
		 = uint64(bigen.Uint64(.b[:8]))
	case 7:
		.d.decRd.readb(.b[:8])
		 = uint64(bigen.Uint64(.b[:8]))
	default:
		.d.errorf("unsigned integers with greater than 64 bits of precision not supported")
		return
	}
	return
}

func ( *bincDecDriver) () ( uint64,  bool) {
	,  := .vd, .vs
	if  == bincVdPosInt {
		 = .decUint()
	} else if  == bincVdNegInt {
		 = .decUint()
		 = true
	} else if  == bincVdSmallInt {
		 = uint64(.vs) + 1
	} else if  == bincVdSpecial {
		if  == bincSpZero {
			//i = 0
		} else if  == bincSpNegOne {
			 = true
			 = 1
		} else {
			.d.errorf("integer decode fails - invalid special value from descriptor %x-%x/%s",
				.vd, .vs, bincdesc(.vd, .vs))
			return
		}
	} else {
		.d.errorf("integer can only be decoded from int/uint. d.bd: 0x%x, d.vd: 0x%x", .bd, .vd)
		return
	}
	return
}

func ( *bincDecDriver) () ( float64) {
	if .advanceNil() {
		return
	}
	,  := .vd, .vs
	if  == bincVdSpecial {
		.bdRead = false
		if  == bincSpNan {
			return math.NaN()
		} else if  == bincSpPosInf {
			return math.Inf(1)
		} else if  == bincSpZeroFloat ||  == bincSpZero {
			return
		} else if  == bincSpNegInf {
			return math.Inf(-1)
		} else {
			.d.errorf("float - invalid special value from descriptor %x-%x/%s",
				.vd, .vs, bincdesc(.vd, .vs))
			return
		}
	} else if  == bincVdFloat {
		 = .decFloat()
	} else {
		 = float64(.DecodeInt64())
	}
	.bdRead = false
	return
}

// bool can be decoded from bool only (single byte).
func ( *bincDecDriver) () ( bool) {
	if .advanceNil() {
		return
	}
	if .bd == (bincVdSpecial | bincSpFalse) {
		// b = false
	} else if .bd == (bincVdSpecial | bincSpTrue) {
		 = true
	} else {
		.d.errorf("bool - %s %x-%x/%s", msgBadDesc, .vd, .vs, bincdesc(.vd, .vs))
		return
	}
	.bdRead = false
	return
}

func ( *bincDecDriver) () ( uint64) {
	if  := .vs;  == 0 {
		 = uint64(.d.decRd.readn1())
	} else if  == 1 {

	} else if  == 2 {

	} else {
		.d.decRd.readb(.b[:8])

	}
	return
}

// EncodeTime encodes a time.Time as a []byte, including
// information on the instant in time and UTC offset.
//
// Format Description
//
//   A timestamp is composed of 3 components:
//
//   - secs: signed integer representing seconds since unix epoch
//   - nsces: unsigned integer representing fractional seconds as a
//     nanosecond offset within secs, in the range 0 <= nsecs < 1e9
//   - tz: signed integer representing timezone offset in minutes east of UTC,
//     and a dst (daylight savings time) flag
//
//   When encoding a timestamp, the first byte is the descriptor, which
//   defines which components are encoded and how many bytes are used to
//   encode secs and nsecs components. *If secs/nsecs is 0 or tz is UTC, it
//   is not encoded in the byte array explicitly*.
//
//       Descriptor 8 bits are of the form `A B C DDD EE`:
//           A:   Is secs component encoded? 1 = true
//           B:   Is nsecs component encoded? 1 = true
//           C:   Is tz component encoded? 1 = true
//           DDD: Number of extra bytes for secs (range 0-7).
//                If A = 1, secs encoded in DDD+1 bytes.
//                    If A = 0, secs is not encoded, and is assumed to be 0.
//                    If A = 1, then we need at least 1 byte to encode secs.
//                    DDD says the number of extra bytes beyond that 1.
//                    E.g. if DDD=0, then secs is represented in 1 byte.
//                         if DDD=2, then secs is represented in 3 bytes.
//           EE:  Number of extra bytes for nsecs (range 0-3).
//                If B = 1, nsecs encoded in EE+1 bytes (similar to secs/DDD above)
//
//   Following the descriptor bytes, subsequent bytes are:
//
//       secs component encoded in `DDD + 1` bytes (if A == 1)
//       nsecs component encoded in `EE + 1` bytes (if B == 1)
//       tz component encoded in 2 bytes (if C == 1)
//
//   secs and nsecs components are integers encoded in a BigEndian
//   2-complement encoding format.
//
//   tz component is encoded as 2 bytes (16 bits). Most significant bit 15 to
//   Least significant bit 0 are described below:
//
//       Timezone offset has a range of -12:00 to +14:00 (ie -720 to +840 minutes).
//       Bit 15 = have\_dst: set to 1 if we set the dst flag.
//       Bit 14 = dst\_on: set to 1 if dst is in effect at the time, or 0 if not.
