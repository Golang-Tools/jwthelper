/*
Msgpack-c implementation powers the c, c++, python, ruby, etc libraries.
We need to maintain compatibility with it and how it encodes integer values
without caring about the type.

For compatibility with behaviour of msgpack-c reference implementation:
  - Go intX (>0) and uintX
       IS ENCODED AS
    msgpack +ve fixnum, unsigned
  - Go intX (<0)
       IS ENCODED AS
    msgpack -ve fixnum, signed
*/

func ( byte) string {
	switch  {
	case mpNil:
		return "nil"
	case mpFalse:
		return "false"
	case mpTrue:
		return "true"
	case mpFloat, mpDouble:
		return "float"
	case mpUint8, mpUint16, mpUint32, mpUint64:
		return "uint"
	case mpInt8, mpInt16, mpInt32, mpInt64:
		return "int"
	default:
		switch {
		case  >= mpPosFixNumMin &&  <= mpPosFixNumMax:
			return "int"
		case  >= mpNegFixNumMin &&  <= mpNegFixNumMax:
			return "int"
		case  == mpStr8,  == mpStr16,  == mpStr32,  >= mpFixStrMin &&  <= mpFixStrMax:
			return "string|bytes"
		case  == mpBin8,  == mpBin16,  == mpBin32:
			return "bytes"
		case  == mpArray16,  == mpArray32,  >= mpFixArrayMin &&  <= mpFixArrayMax:
			return "array"
		case  == mpMap16,  == mpMap32,  >= mpFixMapMin &&  <= mpFixMapMax:
			return "map"
		case  >= mpFixExt1 &&  <= mpFixExt16,  >= mpExt8 &&  <= mpExt32:
			return "ext"
		default:
			return "unknown"
		}
	}
}

// in sequence in the slice.
//
// The Codec then passes it AS-IS to the rpc service (without wrapping it in an

// A MsgpackContainer type specifies the different types of msgpackContainers.
type msgpackContainerType struct {
	fixCutoff             uint8
	bFixMin, b8, b16, b32 byte
	// hasFixMin, has8, has8Always bool
}

var (
	msgpackContainerRawLegacy = msgpackContainerType{
		32, mpFixStrMin, 0, mpStr16, mpStr32,
	}
	msgpackContainerStr = msgpackContainerType{
		32, mpFixStrMin, mpStr8, mpStr16, mpStr32, // true, true, false,
	}
	msgpackContainerBin = msgpackContainerType{
		0, 0, mpBin8, mpBin16, mpBin32, // false, true, true,
	}
	msgpackContainerList = msgpackContainerType{
		16, mpFixArrayMin, 0, mpArray16, mpArray32, // true, false, false,
	}
	msgpackContainerMap = msgpackContainerType{
		16, mpFixMapMin, 0, mpMap16, mpMap32, // true, false, false,
	}
)

type msgpackEncDriver struct {
	noBuiltInTypes
	encDriverNoopContainerWriter
	h *MsgpackHandle
	x [8]byte
	_ [6]uint64 // padding
	e Encoder
}

func ( *msgpackEncDriver) () *Encoder {
	return &.e
}

func ( *msgpackEncDriver) () {
	.e.encWr.writen1(mpNil)
}

func ( *msgpackEncDriver) ( int64) {
	if .h.PositiveIntUnsigned &&  >= 0 {
		.EncodeUint(uint64())
	} else if  > math.MaxInt8 {
		if  <= math.MaxInt16 {
			.e.encWr.writen1(mpInt16)
			bigenHelper{.x[:2], .e.w()}.writeUint16(uint16())
		} else if  <= math.MaxInt32 {
			.e.encWr.writen1(mpInt32)
			bigenHelper{.x[:4], .e.w()}.writeUint32(uint32())
		} else {
			.e.encWr.writen1(mpInt64)
			bigenHelper{.x[:8], .e.w()}.writeUint64(uint64())
		}
	} else if  >= -32 {
		if .h.NoFixedNum {
			.e.encWr.writen2(mpInt8, byte())
		} else {
			.e.encWr.writen1(byte())
		}
	} else if  >= math.MinInt8 {
		.e.encWr.writen2(mpInt8, byte())
	} else if  >= math.MinInt16 {
		.e.encWr.writen1(mpInt16)
		bigenHelper{.x[:2], .e.w()}.writeUint16(uint16())
	} else if  >= math.MinInt32 {
		.e.encWr.writen1(mpInt32)
		bigenHelper{.x[:4], .e.w()}.writeUint32(uint32())
	} else {
		.e.encWr.writen1(mpInt64)
		bigenHelper{.x[:8], .e.w()}.writeUint64(uint64())
	}
}

func ( *msgpackEncDriver) ( uint64) {
	if  <= math.MaxInt8 {
		if .h.NoFixedNum {
			.e.encWr.writen2(mpUint8, byte())
		} else {
			.e.encWr.writen1(byte())
		}
	} else if  <= math.MaxUint8 {
		.e.encWr.writen2(mpUint8, byte())
	} else if  <= math.MaxUint16 {
		.e.encWr.writen1(mpUint16)
		bigenHelper{.x[:2], .e.w()}.writeUint16(uint16())
	} else if  <= math.MaxUint32 {
		.e.encWr.writen1(mpUint32)
		bigenHelper{.x[:4], .e.w()}.writeUint32(uint32())
	} else {
		.e.encWr.writen1(mpUint64)
		bigenHelper{.x[:8], .e.w()}.writeUint64(uint64())
	}
}

func ( *msgpackEncDriver) ( bool) {
	if  {
		.e.encWr.writen1(mpTrue)
	} else {
		.e.encWr.writen1(mpFalse)
	}
}

func ( *msgpackEncDriver) ( float32) {
	.e.encWr.writen1(mpFloat)
	bigenHelper{.x[:4], .e.w()}.writeUint32(math.Float32bits())
}

func ( *msgpackEncDriver) ( float64) {
	.e.encWr.writen1(mpDouble)
	bigenHelper{.x[:8], .e.w()}.writeUint64(math.Float64bits())
}

func ( *msgpackEncDriver) ( time.Time) {
	if .IsZero() {
		.EncodeNil()
		return
	}
	 = .UTC()
	,  := .Unix(), uint64(.Nanosecond())
	var  uint64
	var  = 4
	if  >= 0 && >>34 == 0 {
		 = ( << 34) | uint64()
		if &0xffffffff00000000 != 0 {
			 = 8
		}
	} else {
		 = 12
	}
	if .h.WriteExt {
		.encodeExtPreamble(mpTimeExtTagU, )

func ( *msgpackEncDriver) ( interface{},  uint64,  Ext) {
	var  []byte
	if  == SelfExt {
		 = .e.blist.get(1024)[:0]
		.e.sideEncode(, &)
	} else {

	}

		return

func ( *msgpackEncDriver) ( byte,  int) {
	if  == 1 {
		.e.encWr.writen2(mpFixExt1, )
	} else if  == 2 {
		.e.encWr.writen2(mpFixExt2, )
	} else if  == 4 {
		.e.encWr.writen2(mpFixExt4, )
	} else if  == 8 {
		.e.encWr.writen2(mpFixExt8, )
	} else if  == 16 {
		.e.encWr.writen2(mpFixExt16, )
	} else if  < 256 {
		.e.encWr.writen2(mpExt8, byte())
		.e.encWr.writen1()
	} else if  < 65536 {
		.e.encWr.writen1(mpExt16)
		bigenHelper{.x[:2], .e.w()}.writeUint16(uint16())
		.e.encWr.writen1()
	} else {
		.e.encWr.writen1(mpExt32)
		bigenHelper{.x[:4], .e.w()}.writeUint32(uint32())
		.e.encWr.writen1()
	}
}

func ( *msgpackEncDriver) ( int) {
	.writeContainerLen(msgpackContainerList, )
}

func ( *msgpackEncDriver) ( int) {
	.writeContainerLen(msgpackContainerMap, )
}

func ( *msgpackEncDriver) ( string) {
	var  msgpackContainerType
	if .h.WriteExt {
		if .h.StringToRaw {
			 = msgpackContainerBin
		} else {
			 = msgpackContainerStr
		}
	} else {
		 = msgpackContainerRawLegacy
	}
	.writeContainerLen(, len())
	if len() > 0 {
		.e.encWr.writestr()
	}
}

func ( *msgpackEncDriver) ( []byte) {
	if  == nil {
		.EncodeNil()
		return
	}
	if .h.WriteExt {
		.writeContainerLen(msgpackContainerBin, len())
	} else {
		.writeContainerLen(msgpackContainerRawLegacy, len())
	}
	if len() > 0 {
		.e.encWr.writeb()
	}
}

func ( *msgpackEncDriver) ( msgpackContainerType,  int) {
	if .fixCutoff > 0 &&  < int(.fixCutoff) {
		.e.encWr.writen1(.bFixMin | byte())
	} else if .b8 > 0 &&  < 256 {
		.e.encWr.writen2(.b8, uint8())
	} else if  < 65536 {
		.e.encWr.writen1(.b16)
		bigenHelper{.x[:2], .e.w()}.writeUint16(uint16())
	} else {
		.e.encWr.writen1(.b32)
		bigenHelper{.x[:4], .e.w()}.writeUint32(uint32())
	}
}

type msgpackDecDriver struct {
	decDriverNoopContainerReader
	h *MsgpackHandle
	// b      [scratchByteArrayLen]byte
	bd     byte
	bdRead bool
	fnil   bool
	noBuiltInTypes
	_ [6]uint64 // padding
	d Decoder
}

func ( *msgpackDecDriver) () *Decoder {
	return &.d
}

// Note: This returns either a primitive (int, bool, etc) for non-containers,
// or a containerType, or a specific type denoting nil or extension.
// It is called when a nil interface{} is passed, leaving it up to the DecDriver
// to introspect the stream and decide how best to decode.
// It deciphers the value by looking at the stream first.
func ( *msgpackDecDriver) () {
	if !.bdRead {
		.readNextBd()
	}
	.fnil = false
	 := .bd
	 := .d.naked()
	var  bool

	switch  {
	case mpNil:
		.v = valueTypeNil
		.bdRead = false
		.fnil = true
	case mpFalse:
		.v = valueTypeBool
		.b = false
	case mpTrue:
		.v = valueTypeBool
		.b = true

	case mpFloat:
		.v = valueTypeFloat
		.f = float64(math.Float32frombits(bigen.Uint32(.d.decRd.readx(4))))
	case mpDouble:
		.v = valueTypeFloat
		.f = math.Float64frombits(bigen.Uint64(.d.decRd.readx(8)))

	case mpUint8:
		.v = valueTypeUint
		.u = uint64(.d.decRd.readn1())
	case mpUint16:
		.v = valueTypeUint
		.u = uint64(bigen.Uint16(.d.decRd.readx(2)))
	case mpUint32:
		.v = valueTypeUint
		.u = uint64(bigen.Uint32(.d.decRd.readx(4)))
	case mpUint64:
		.v = valueTypeUint
		.u = uint64(bigen.Uint64(.d.decRd.readx(8)))

	case mpInt8:
		.v = valueTypeInt
		.i = int64(int8(.d.decRd.readn1()))
	case mpInt16:
		.v = valueTypeInt
		.i = int64(int16(bigen.Uint16(.d.decRd.readx(2))))
	case mpInt32:
		.v = valueTypeInt
		.i = int64(int32(bigen.Uint32(.d.decRd.readx(4))))
	case mpInt64:
		.v = valueTypeInt
		.i = int64(int64(bigen.Uint64(.d.decRd.readx(8))))

	default:
		switch {
		case  >= mpPosFixNumMin &&  <= mpPosFixNumMax:
			// positive fixnum (always signed)
			.v = valueTypeInt
			.i = int64(int8())
		case  >= mpNegFixNumMin &&  <= mpNegFixNumMax:
			// negative fixnum
			.v = valueTypeInt
			.i = int64(int8())
		case  == mpStr8,  == mpStr16,  == mpStr32,  >= mpFixStrMin &&  <= mpFixStrMax:
			if .h.WriteExt || .h.RawToString {
				.v = valueTypeString
				.s = string(.DecodeStringAsBytes())
			} else {
				.v = valueTypeBytes
				.l = .DecodeBytes(nil, false)
			}
		case  == mpBin8,  == mpBin16,  == mpBin32:
			decNakedReadRawBytes(, &.d, , .h.RawToString)
		case  == mpArray16,  == mpArray32,  >= mpFixArrayMin &&  <= mpFixArrayMax:
			.v = valueTypeArray
			 = true
		case  == mpMap16,  == mpMap32,  >= mpFixMapMin &&  <= mpFixMapMax:
			.v = valueTypeMap
			 = true
		case  >= mpFixExt1 &&  <= mpFixExt16,  >= mpExt8 &&  <= mpExt32:
			.v = valueTypeExt
			 := .readExtLen()
			.u = uint64(.d.decRd.readn1())
			if .u == uint64(mpTimeExtTagU) {
				.v = valueTypeTime
				.t = .decodeTime()
			} else if .d.bytes {
				.l = .d.decRd.readx(uint())
			} else {
				.l = decByteSlice(.d.r(), , .d.h.MaxInitLen, .d.b[:])
			}
		default:
			.d.errorf("cannot infer value: %s: Ox%x/%d/%s", msgBadDesc, , , mpdesc())
		}
	}
	if ! {
		.bdRead = false
	}
	if .v == valueTypeUint && .h.SignedInteger {
		.v = valueTypeInt
		.i = int64(.u)
	}
}

// int can be decoded from msgpack type: intXXX or uintXXX
func ( *msgpackDecDriver) () ( int64) {
	if .advanceNil() {
		return
	}
	switch .bd {
	case mpUint8:
		 = int64(uint64(.d.decRd.readn1()))
	case mpUint16:
		 = int64(uint64(bigen.Uint16(.d.decRd.readx(2))))
	case mpUint32:
		 = int64(uint64(bigen.Uint32(.d.decRd.readx(4))))
	case mpUint64:
		 = int64(bigen.Uint64(.d.decRd.readx(8)))
	case mpInt8:
		 = int64(int8(.d.decRd.readn1()))
	case mpInt16:
		 = int64(int16(bigen.Uint16(.d.decRd.readx(2))))
	case mpInt32:
		 = int64(int32(bigen.Uint32(.d.decRd.readx(4))))
	case mpInt64:
		 = int64(bigen.Uint64(.d.decRd.readx(8)))
	default:
		switch {
		case .bd >= mpPosFixNumMin && .bd <= mpPosFixNumMax:
			 = int64(int8(.bd))
		case .bd >= mpNegFixNumMin && .bd <= mpNegFixNumMax:
			 = int64(int8(.bd))
		default:
			.d.errorf("cannot decode signed integer: %s: %x/%s", msgBadDesc, .bd, mpdesc(.bd))
			return
		}
	}
	.bdRead = false
	return
}

// uint can be decoded from msgpack type: intXXX or uintXXX
func ( *msgpackDecDriver) () ( uint64) {
	if .advanceNil() {
		return
	}
	switch .bd {
	case mpUint8:
		 = uint64(.d.decRd.readn1())
	case mpUint16:
		 = uint64(bigen.Uint16(.d.decRd.readx(2)))
	case mpUint32:
		 = uint64(bigen.Uint32(.d.decRd.readx(4)))
	case mpUint64:
		 = bigen.Uint64(.d.decRd.readx(8))
	case mpInt8:
		if  := int64(int8(.d.decRd.readn1()));  >= 0 {
			 = uint64()
		} else {
			.d.errorf("assigning negative signed value: %v, to unsigned type", )
			return
		}
	case mpInt16:
		if  := int64(int16(bigen.Uint16(.d.decRd.readx(2))));  >= 0 {
			 = uint64()
		} else {
			.d.errorf("assigning negative signed value: %v, to unsigned type", )
			return
		}
	case mpInt32:
		if  := int64(int32(bigen.Uint32(.d.decRd.readx(4))));  >= 0 {
			 = uint64()
		} else {
			.d.errorf("assigning negative signed value: %v, to unsigned type", )
			return
		}
	case mpInt64:
		if  := int64(bigen.Uint64(.d.decRd.readx(8)));  >= 0 {
			 = uint64()
		} else {
			.d.errorf("assigning negative signed value: %v, to unsigned type", )
			return
		}
	default:
		switch {
		case .bd >= mpPosFixNumMin && .bd <= mpPosFixNumMax:
			 = uint64(.bd)
		case .bd >= mpNegFixNumMin && .bd <= mpNegFixNumMax:
			.d.errorf("assigning negative signed value: %v, to unsigned type", int(.bd))
			return
		default:
			.d.errorf("cannot decode unsigned integer: %s: %x/%s", msgBadDesc, .bd, mpdesc(.bd))
			return
		}
	}
	.bdRead = false
	return
}

// float can either be decoded from msgpack type: float, double or intX
func ( *msgpackDecDriver) () ( float64) {
	if .advanceNil() {
		return
	}
	if .bd == mpFloat {
		 = float64(math.Float32frombits(bigen.Uint32(.d.decRd.readx(4))))
	} else if .bd == mpDouble {
		 = math.Float64frombits(bigen.Uint64(.d.decRd.readx(8)))
	} else {
		 = float64(.DecodeInt64())
	}
	.bdRead = false
	return
}

// bool can be decoded from bool, fixnum 0 or 1.
func ( *msgpackDecDriver) () ( bool) {
	if .advanceNil() {
		return
	}
	if .bd == mpFalse || .bd == 0 {
		// b = false
	} else if .bd == mpTrue || .bd == 1 {
		 = true
	} else {
		.d.errorf("cannot decode bool: %s: %x/%s", msgBadDesc, .bd, mpdesc(.bd))
		return
	}
	.bdRead = false
	return
}

func ( *msgpackDecDriver) ( []byte,  bool) ( []byte) {
	if .advanceNil() {
		return
	}

	 := .bd
	var  int
	if  == mpBin8 ||  == mpBin16 ||  == mpBin32 {
		 = .readContainerLen(msgpackContainerBin) // binary
	} else if  == mpStr8 ||  == mpStr16 ||  == mpStr32 ||
		( >= mpFixStrMin &&  <= mpFixStrMax) {
		 = .readContainerLen(msgpackContainerStr) // string/raw
	} else if  == mpArray16 ||  == mpArray32 ||
		( >= mpFixArrayMin &&  <= mpFixArrayMax) {
		// check if an "array" of uint8's
		if  && len() == 0 {
			 = .d.b[:]
		}
		// bsOut, _ = fastpathTV.DecSliceUint8V(bs, true, d.d)
		 := .ReadArrayStart()
		 = usableByteSlice(, )
		for  := 0;  < len(); ++ {
			[] = uint8(chkOvf.UintV(.DecodeUint64(), 8))
		}
		return 
	} else {
		.d.errorf("invalid byte descriptor for decoding bytes, got: 0x%x", .bd)
		return
	}

	.bdRead = false
	if  {
