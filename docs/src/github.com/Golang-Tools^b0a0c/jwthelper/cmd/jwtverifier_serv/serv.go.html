//PerformanceOpts 配置性能调优设置
func ( *Server) () {
	if .opts == nil {
		.opts = []grpc.ServerOption{}
	}

	if .Max_Recv_Msg_Size != 0 {
		.opts = append(.opts, grpc.MaxRecvMsgSize(.Max_Recv_Msg_Size))
	}
	if .Max_Send_Msg_Size != 0 {
		.opts = append(.opts, grpc.MaxSendMsgSize(.Max_Send_Msg_Size))
	}
	if .Initial_Window_Size != 0 {
		.opts = append(.opts, grpc.InitialWindowSize(int32(.Initial_Window_Size)))
	}
	if .Initial_Conn_Window_Size != 0 {
		.opts = append(.opts, grpc.InitialConnWindowSize(int32(.Initial_Conn_Window_Size)))
	}
	if .Max_Concurrent_Streams != 0 {
		.opts = append(.opts, grpc.MaxConcurrentStreams(uint32(.Max_Concurrent_Streams)))
	}
	if .Max_Connection_Idle != 0 || .Max_Connection_Age != 0 || .Max_Connection_Age_Grace != 0 || .Keepalive_Time != 0 || .Keepalive_Timeout != 0 {
		 := keepalive.ServerParameters{
			MaxConnectionIdle:     time.Duration(.Max_Connection_Idle) * time.Second,
			MaxConnectionAge:      time.Duration(.Max_Connection_Age) * time.Second,
			MaxConnectionAgeGrace: time.Duration(.Max_Connection_Age_Grace) * time.Second,
			Time:                  time.Duration(.Keepalive_Time) * time.Second,
			Timeout:               time.Duration(.Keepalive_Timeout) * time.Second,
		}
		.opts = append(.opts, grpc.KeepaliveParams())
	}

	if .Keepalive_Enforcement_Min_Time != 0 || .Keepalive_Enforcement_Permit_Without_Stream {
		 := keepalive.EnforcementPolicy{
			MinTime:             time.Duration(.Keepalive_Enforcement_Min_Time) * time.Second,
			PermitWithoutStream: .Keepalive_Enforcement_Permit_Without_Stream,
		}
		.opts = append(.opts, grpc.KeepaliveEnforcementPolicy())
	}
}
