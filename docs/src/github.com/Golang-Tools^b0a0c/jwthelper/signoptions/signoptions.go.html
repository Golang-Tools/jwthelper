//SignOptions 签名函数参数
type SignOptions struct {
	Sub        string
	Aud        []string
	Exp        int64 // 过期时间戳
	Nbf        int64
	Jti        string
	RefreshExp int64 // >0时会设置refresh_token,其过期时间就是这个字段控制,这个字段也是时间戳
}

var DefaultSignOptions = SignOptions{}

type SignOption interface {
	Apply(*SignOptions)
}

// func (emptyOption) apply(*SignOptions) {}
type funcSignOption struct {
	f func(*SignOptions)
}

func ( *funcSignOption) ( *SignOptions) {
	.f()
}

func ( func(*SignOptions)) *funcSignOption {
	return &funcSignOption{
		f: ,
	}
}
