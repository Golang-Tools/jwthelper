	// 性能设置

	Max_Send_Msg_Size                           int  `json:"max_send_msg_size,omitempty" jsonschema:"description=允许发送的最大消息长度"`

//initKeepalive 初始化keepalive的相关设置
func ( *SDK) () {
	if .Keepalive_Time != 0 || .Keepalive_Timeout != 0 || .Keepalive_Enforcement_Permit_Without_Stream {
		 := keepalive.ClientParameters{
			Time:                time.Duration(.Keepalive_Time) * time.Second,
			Timeout:             time.Duration(.Keepalive_Timeout) * time.Second,
			PermitWithoutStream: .Keepalive_Enforcement_Permit_Without_Stream, // send pings even without active streams
		}
		.opts = append(.opts, grpc.WithKeepaliveParams())
	}
}

//initPerformanceOpts 初始化连接的性能选项
func ( *SDK) () {
	if .Conn_With_Block {
		.opts = append(.opts, grpc.WithBlock())
	}
	if .Initial_Window_Size != 0 {
		.opts = append(.opts, grpc.WithInitialWindowSize(int32(.Initial_Window_Size)))
	}
	if .Initial_Conn_Window_Size != 0 {
		.opts = append(.opts, grpc.WithInitialConnWindowSize(int32(.Initial_Conn_Window_Size)))
	}
}

				RootCAs:      ,
