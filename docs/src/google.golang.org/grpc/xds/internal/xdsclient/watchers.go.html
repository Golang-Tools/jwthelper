type watchInfoState int

const (
	watchInfoStateStarted watchInfoState = iota
	watchInfoStateRespReceived
	watchInfoStateTimeout
	watchInfoStateCanceled
)

// watchInfo holds all the information from a watch() call.
type watchInfo struct {
	c      *clientImpl
	rType  ResourceType
	target string

	ldsCallback func(ListenerUpdate, error)
	rdsCallback func(RouteConfigUpdate, error)
	cdsCallback func(ClusterUpdate, error)
	edsCallback func(EndpointsUpdate, error)

	expiryTimer *time.Timer

	// mu protects state, and c.scheduleCallback().
	// - No callback should be scheduled after watchInfo is canceled.
	// - No timeout error should be scheduled after watchInfo is resp received.
	mu    sync.Mutex
	state watchInfoState
}

func ( *watchInfo) ( interface{}) {
	.mu.Lock()
	defer .mu.Unlock()
	if .state == watchInfoStateCanceled {
		return
	}
	.state = watchInfoStateRespReceived
	.expiryTimer.Stop()
	.c.scheduleCallback(, , nil)
}

func ( *watchInfo) ( error) {
	.mu.Lock()
	defer .mu.Unlock()
	if .state == watchInfoStateCanceled {
		return
	}
	.state = watchInfoStateRespReceived
	.expiryTimer.Stop()
	.sendErrorLocked()
}

func ( *watchInfo) () {
	.mu.Lock()
	defer .mu.Unlock()
	if .state == watchInfoStateCanceled {
		return
	}
	.state = watchInfoStateRespReceived
	.expiryTimer.Stop()
	.sendErrorLocked(NewErrorf(ErrorTypeResourceNotFound, "xds: %v target %s not found in received response", .rType, .target))
}

func ( *watchInfo) () {
	.mu.Lock()
	defer .mu.Unlock()
	if .state == watchInfoStateCanceled || .state == watchInfoStateRespReceived {
		return
	}
	.state = watchInfoStateTimeout
	.sendErrorLocked(fmt.Errorf("xds: %v target %s not found, watcher timeout", .rType, .target))
}

func ( *watchInfo) () {
	.mu.Lock()
	defer .mu.Unlock()
	if .state == watchInfoStateCanceled {
		return
	}
	.expiryTimer.Stop()
	.state = watchInfoStateCanceled
}

	case ClusterResource:
		 = .cdsWatchers
		 = .cdsMD
	case EndpointsResource:
		 = .edsWatchers
		 = .edsMD

		.logger.Errorf("unknown watch type: %v", .rType)

		// underlying versioned apiClient.
		.logger.Debugf("first watch for type %v, resource name %v, will send a new xDS request", .rType, .target)

	}

	// call for new responses.
	[] = true

		}
	case ClusterResource:

			.newUpdate()
		}
	case EndpointsResource:
		if ,  := .edsCache[];  {
			.logger.Debugf("EDS resource with name %v found in cache: %+v", .target, pretty.ToJSON())
			.newUpdate()
